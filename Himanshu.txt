<!DOCTYPE html>
<html>
<head>
  <title>Display XML using XSL</title>
  <script>
    function displayXML() {
      var xhttp = new XMLHttpRequest();
      xhttp.onreadystatechange = function() {
        if (this.readyState == 4 && this.status == 200) {
          var xml = this.responseXML;
          var xsl = new XMLHttpRequest();
          xsl.onreadystatechange = function() {
            if (this.readyState == 4 && this.status == 200) {
              var xsltProcessor = new XSLTProcessor();
              xsltProcessor.importStylesheet(this.responseXML);
              var resultDocument = xsltProcessor.transformToDocument(xml);
              document.getElementById("content").innerHTML = new XMLSerializer().serializeToString(resultDocument);
            }
          };
          xsl.open("GET", "style.xsl", true);
          xsl.send();
        }
      };
      xhttp.open("GET", "data.xml", true);
      xhttp.send();
    }
  </script>
</head>
<body onload="displayXML()">
  <div id="content"></div>
</body>
</html>
